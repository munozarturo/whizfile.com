## How To

### Standard Format

The standard format covers imports, environment variables

```TypeScript
// the below imporst are universal in all routes
import { handleResponse } from "@/lib/api/utils";
import * as zod from "zod";
import { NextRequest, NextResponse } from "next/server";
import { Collection, Collections, connectToDatabase } from "@/lib/db/mongo";

// the below imports are specific to every route
import { CollectionSchema } from "@/lib/db/schema/schema-path.ts";
import { RequestsReq } from "@/lib/api/validations/api-endpoint-validations.ts";

// other imports
import ... from "..."

// environment variables
if (!process.env.ENV_VAR) {
    throw new Error(
        "`ENV_VAR` environment variable is not defined."
    );
}

const ENV_VAR = process.env.ENV_VAR;

// Next.js Methods

export [async] function GET(...) ...

export [async] function POST(...) ...

export [async] function DELETE(...) ...

export [async] function PUT(...) ...

export [async] function PATCH(...) ...

export [async] function HEAD(...) ...

```

### POST

```TypeScript
export async function POST(req: NextRequest) {
    try {
        // parse input
        const body: zod.infer<typeof RequestsReq> = RequestsReq.parse(
            await req.json()
        );

        // connect to db
        const collections: Collections = await connectToDatabase();
        const collection: Collection<zod.infer<typeof CollectionSchema>> =
            collections.someCollection;

        // api endpoint body

        // standard response
        return NextResponse.json(
            handleResponse("Response message.", { ... }), { status: 200 }
        );
    } catch (e: any) {
        return NextResponse.json(handleError(e));
    }
}
```

Witch context parameters

```TypeScript
export async function POST(req: NextRequest, context: { readonly params: { readonly someParameter: string } }) ...
```

Preprocessing (eg. Auth)

```TypeScript
export async function POST(req: NextRequest) {
    // preliminary checks like authorization
    if (req.headers.get("Authorization") !== SECRET_KEY) {
        return NextResponse.json(handleResponse("Unathorized"), {
            status: 401,
        });
    }

    // rest of body
    try {
        ...
    } catch (...) {
        ...
    }
}
```
